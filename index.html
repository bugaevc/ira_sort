IraSort

IraSort, named after my awesome friend Ira, sometimes referred to as insertion sort, is a quadratic sorting algorithm that is easy to implement and comprehend.

Implementations

Python

def ira_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(i+1, n):
            if arr[i] > arr[j]:
                arr[i], arr[j] = arr[j], arr[i]

Rust

fn ira_sort<T: Ord>(arr: &mut [T]) {
    let n = arr.len();
    for i in 0..n {
        for j in i+1..n {
            if arr[i] > arr[j] {
                arr.swap(i, j);
            }
        }
    }
}

C++

#include <vector>

template <typename T>
void ira_sort(std::vector<T> &arr) {
    size_t n = arr.size();
    for (size_t i = 0; i < n; i++) {
        for (size_t j = i+1; j < n; j++) {
            if (arr[i] > arr[j]) {
                std::swap(arr[i], arr[j]);
            }
        }
    }
}
